    \item

    \lstinline{#body img}

    \item
    \lstinline{h1 div}

    \item
    \lstinline{h2 + li}

    \item
    \lstinline{img ~ li > em}

    \item
    \lstinline{#header > h1 ~ *}

    \item

    \item

    \item
    \lstinline{ol > em}

    \lstinline{[src="ceasar.png"] ~ *}

    \item

    \item
    \lstinline{div + *, h2}
